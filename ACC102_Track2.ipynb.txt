# ACC102 Mini Assignment Track 2 - （comp.funda）
import wrds
import pandas as pd
# 1. Connect to WRDS (replace with your username)
db = wrds.Connection(wrds_username='Your WRDS username')
# 2. Use the comp.funda table (all students have default access) to query data of major US consumer electronics companies
# Select Apple, Samsung Electronics, Sony, these are all must-have data on WRDS
query = """
SELECT gvkey, fyear, ni, revt, at, conm
FROM comp.funda
WHERE conm IN ('APPLE INC', 'SAMSUNG ELECTRONICS CO LTD', 'SONY GROUP CORP')
  AND indfmt = 'INDL' 
  AND datafmt = 'STD' 
  AND popsrc = 'D' 
  AND consol = 'C'
  AND fyear BETWEEN 2018 AND 2024
ORDER BY conm, fyear
"""
# Execute query
df = db.raw_sql(query)
db.close()  
# 3. Data cleaning (deleting missing values)
df = df.dropna()
# 4. Calculate core financial indicators
df['profit_margin'] = df['ni'] / df['revt']  # Net Profit Margin = Net Profit / Revenue
df['roe'] = df['ni'] / df['at']               # ROE = ROE = Net Profit / Total Assets
df['asset_turnover'] = df['revt'] / df['at']  # Asset Turnover = Revenue / Total Assets
# 5. Export Excel
df.to_excel('consumer_electronics_financials.xlsx', index=False)
# 6. Print result verification
print("✅ Code execution completed！No errors reported！The Excel file has been generated！")
print("\nPreview of Financial Indicator Results：")
print(df[['conm','fyear','profit_margin','roe','asset_turnover']].round(4))
